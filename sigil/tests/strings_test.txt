## Sources

src h : 'hello'
src w : "world"
src s : " "

src helloWorld : ""
src helloWorld2 : ""


## Sigils

# Is entered first that concats to make hello world
sigil HelloWorldConcat ? h and w = "world":
    helloWorld : h + s + w

# Is entered second
sigil HelloWorld2InitSet ? h and helloWorld2 = "":
    helloWorld2 : helloWorld
    invoke LoopS

# Is entered third to recursively increase s
sigil LoopS ? s != "    ":
    s : s + " "
    invoke Pulse, HelloWorldNext

# Is entered fourth that makes the final string of helloWorld2
src i : "2"
sigil HelloWorldNext:
    helloWorld2 : s + helloWorld2 + i
    invoke HelloWorldPrint

# Is entered fifth to invoke Whisper which implicitly passes the args in the conditional
sigil HelloWorldPrint ? helloWorld and helloWorld2:
    invoke Whisper


## Run

invoke h

# Output: hello world    hello world2
# Runtime chain: ['h', 'HelloWorldConcat', 'HelloWorld2InitSet', 'LoopS', 'Pulse', 'LoopS', \
# 'Pulse', 'LoopS', 'Pulse', 'HelloWorldNext', 'HelloWorldPrint', 'Whisper']
